```javascript
// 在 DOMContentLoaded 或页面初始化时添加
document.addEventListener('DOMContentLoaded', function() {
    // 1. 获取当前孩子信息
    if (window.Android && Android.getCurrentChildId) {
        const childId = Android.getCurrentChildId();
        if (childId && childId !== '-1') {
            // 2. 获取孩子详情
            const childJson = Android.getChildProfileById(childId);
            const child = JSON.parse(childJson);
            
            // 3. 自动填充表单
            if (child.name) {
                document.getElementById('childNameInput').value = child.name;
            }
            if (child.birthDate) {
                document.getElementById('childAgeInput').value = calculateAge(child.birthDate);
            }
            if (child.gender) {
                document.getElementById('genderSelect').value = child.gender === 'BOY' ? '男孩' : '女孩';
            }
            
            // 4. 设置只读或标记为"已自动填充"
            document.getElementById('childNameInput').readonly = true;
            document.getElementById('childNameInput').style.background = '#f5f5f5';
        }
    }
});

function calculateAge(birthDate) {
    const birth = new Date(birthDate);
    const today = new Date();
    let age = today.getFullYear() - birth.getFullYear();
    const monthDiff = today.getMonth() - birth.getMonth();
    if (monthDiff < 0 || (monthDiff === 0 && today.getDate() < birth.getDate())) {
        age--;
    }
    return age;
}
```

**Service 层：**

```java
public Page<ConsultationRecord> getUserRecords(
        Long userId, Long childId, String startDate, String endDate, Pageable pageable) {
    
    Specification<ConsultationRecord> spec = Specification.where(
            (root, query, cb) -> cb.equal(root.get("parentUserId"), userId)
    );
    
    if (childId != null) {
        spec = spec.and((root, query, cb) -> cb.equal(root.get("childId"), childId));
    }
    
    if (startDate != null) {
        spec = spec.and((root, query, cb) -> 
                cb.greaterThanOrEqualTo(root.get("createdAt"), LocalDateTime.parse(startDate)));
    }
    
    if (endDate != null) {
        spec = spec.and((root, query, cb) -> 
                cb.lessThanOrEqualTo(root.get("createdAt"), LocalDateTime.parse(endDate)));
    }
    
    return consultationRecordRepository.findAll(spec, pageable);
}
```

```java
@Transactional
public UserBadge awardBadge(Long userId, String badgeId, String category) {
    // 1. 创建徽章记录
    UserBadge badge = new UserBadge();
    badge.setUserId(userId);
    badge.setBadgeId(badgeId);
    badge.setBadgeName(BadgeType.valueOf(badgeId.toUpperCase()).getName());
    badge.setCategory(category);
    badge.setEarnedAt(LocalDateTime.now());
    badge = userBadgeRepository.save(badge);
    
    // 2. 如果是智慧妈妈徽章，升级为 GOLD 咨询师
    if ("wisdom_mom".equals(badgeId)) {
        Consultant consultant = consultantRepository.findByUserId(userId)
                .orElseGet(() -> createConsultantFromParent(userId));
        
        consultant.setIdentityTier(IdentityTier.GOLD);
        consultant.setSpecialty(category);  // 可指导的领域
        consultantRepository.save(consultant);
    }
    
    return badge;
}

private Consultant createConsultantFromParent(Long userId) {
    User user = userRepository.findById(userId).orElseThrow();
    Consultant consultant = new Consultant();
    consultant.setUserId(userId);
    consultant.setName(user.getNickname() != null ? user.getNickname() : "指导师");
    consultant.setIdentityTier(IdentityTier.BRONZE);
    return consultantRepository.save(consultant);
}
```
